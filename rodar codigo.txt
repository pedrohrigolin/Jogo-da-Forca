Rodando Código via Terminal

# Criar pasta build
mkdir build

cd build

# Gere com MinGW
cmake .. -G "MinGW Makefiles"

# Compile
cmake --build .

# Rode o executável
.\..\bin\JogoDaForca.exe



Caso já tenha criado antes os arquivo cmake
# Limpa cache antigo (IMPORTANTE!)
Remove-Item -Recurse -Force CMakeCache.txt, CMakeFiles
